Lab 05: Sparse Autoencoder Walkthrough

Track 4: Code Labs
05 of 05
~90 min
GPT-2 small + custom SAE

This lab trains a complete Sparse Autoencoder on GPT-2 MLP layer 9 activations. You will collect a corpus of activations, implement the SAE with proper normalisation and loss, train it, evaluate reconstruction quality, inspect the discovered features, and perform feature steering to confirm causal influence.

Compute note

Training a full SAE requires a large activation corpus. This lab uses a small corpus (5000 token positions) which is sufficient for demonstration but will not produce publication-quality features. Plan for ~10 minutes training time on CPU. With GPU it takes ~2 minutes.

Setup

import torch
import torch.nn as nn
import numpy as np
import matplotlib.pyplot as plt
from transformers import GPT2LMHeadModel, GPT2Tokenizer
from torch.utils.data import DataLoader, TensorDataset

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
print(f"Device: {device}")

model = GPT2LMHeadModel.from_pretrained("gpt2").to(device)
tok   = GPT2Tokenizer.from_pretrained("gpt2")
model.eval()

D_MODEL  = 768   # GPT-2 hidden dim
D_SAE    = 768 * 4  # SAE expansion factor 4x
LAYER    = 9      # MLP layer to study
L1_COEFF = 1e-3   # sparsity coefficient

Step 1: Collect MLP Activations

# A small corpus of diverse text
corpus_texts = [
    "The capital of France is Paris and the Eiffel Tower stands there.",
    "Scientists have discovered a new exoplanet in the habitable zone.",
    "The stock market fell sharply following the central bank announcement.",
    "She played Beethoven's Moonlight Sonata with remarkable expression.",
    "DNA replication occurs during the S phase of the cell cycle.",
    "The algorithm runs in O(n log n) time and uses O(n) space.",
    "Leonardo da Vinci painted the Mona Lisa in the early 16th century.",
    "The transformer model uses self-attention to process sequences.",
    "Photosynthesis converts carbon dioxide and water into glucose.",
    "The Great Wall of China stretches over 13,000 miles.",
] * 20  # repeat for more samples

activation_buffer = []

def save_mlp_output(module, inp, out):
    activation_buffer.append(out.detach().cpu())  # (1, T, 768)

hook = model.transformer.h[LAYER].mlp.register_forward_hook(save_mlp_output)

for text in corpus_texts:
    ids = tok(text, return_tensors="pt", max_length=64, truncation=True).to(device)
    with torch.no_grad():
        model(**ids)

hook.remove()

# Flatten all token positions into one big matrix
all_acts = torch.cat([a.reshape(-1, D_MODEL) for a in activation_buffer], dim=0)
print(f"Activation corpus: {all_acts.shape[0]} token positions × {D_MODEL} dims")
print(f"Mean norm: {all_acts.norm(dim=-1).mean():.3f}")
print(f"Std norm:  {all_acts.norm(dim=-1).std():.3f}")

Step 2: SAE Architecture

class SparseAutoencoder(nn.Module):
    """
    A minimal but correct SAE following Bricken et al. 2023.
    Key details:
      - b_pre centres the input distribution before encoding
      - W_dec columns are constrained to unit norm (features live on sphere)
      - Loss = reconstruction MSE + L1 sparsity penalty on hidden activations
    """
    def __init__(self, d_model, d_hidden, l1_coeff=1e-3):
        super().__init__()
        self.d_model  = d_model
        self.d_hidden = d_hidden
        self.l1_coeff = l1_coeff

        self.b_pre   = nn.Parameter(torch.zeros(d_model))
        self.W_enc   = nn.Parameter(torch.randn(d_model, d_hidden) * (1 / d_model**0.5))
        self.b_enc   = nn.Parameter(torch.zeros(d_hidden))
        self.W_dec   = nn.Parameter(torch.randn(d_hidden, d_model) * (1 / d_hidden**0.5))
        self.b_dec   = nn.Parameter(torch.zeros(d_model))

        # Initialise decoder columns to unit norm
        with torch.no_grad():
            self.W_dec.data = nn.functional.normalize(self.W_dec.data, dim=1)

    def encode(self, x):
        x_centred = x - self.b_pre
        h = torch.relu(x_centred @ self.W_enc + self.b_enc)  # (B, d_hidden)
        return h

    def decode(self, h):
        return h @ self.W_dec + self.b_dec  # (B, d_model)

    def forward(self, x):
        h     = self.encode(x)
        x_hat = self.decode(h)
        recon = (x - x_hat).pow(2).sum(dim=-1).mean()  # mean over batch
        l1    = self.l1_coeff * h.abs().sum(dim=-1).mean()
        loss  = recon + l1
        return x_hat, h, loss, recon, l1

    def normalise_decoder(self):
        """Call after each optimiser step to keep W_dec columns on unit sphere."""
        with torch.no_grad():
            self.W_dec.data = nn.functional.normalize(self.W_dec.data, dim=1)

sae = SparseAutoencoder(D_MODEL, D_SAE, L1_COEFF).to(device)
n_params = sum(p.numel() for p in sae.parameters())
print(f"SAE parameters: {n_params:,}")
print(f"  Encoder:  {D_MODEL} → {D_SAE}")
print(f"  Decoder:  {D_SAE} → {D_MODEL}")
print(f"  Expansion factor: {D_SAE // D_MODEL}×")

Step 3: Train the SAE

# DataLoader over collected activations
dataset    = TensorDataset(all_acts.to(device))
dataloader = DataLoader(dataset, batch_size=256, shuffle=True)

optimiser  = torch.optim.Adam(sae.parameters(), lr=1e-3)
N_EPOCHS   = 30

history = {"loss": [], "recon": [], "l1": [], "l0": [], "dead_pct": []}
feature_activation_counts = torch.zeros(D_SAE, device=device)

for epoch in range(N_EPOCHS):
    epoch_loss = epoch_recon = epoch_l1 = epoch_l0 = 0.0
    n_batches  = 0

    for (batch,) in dataloader:
        optimiser.zero_grad()
        x_hat, h, loss, recon, l1 = sae(batch)
        loss.backward()
        optimiser.step()
        sae.normalise_decoder()   # keep decoder columns unit-normed

        # Track which features activated this batch
        feature_activation_counts += (h > 0).float().sum(dim=0)

        # Metrics
        l0 = (h > 0).float().sum(dim=-1).mean().item()
        epoch_loss  += loss.item()
        epoch_recon += recon.item()
        epoch_l1    += l1.item()
        epoch_l0    += l0
        n_batches   += 1

    avg_l0   = epoch_l0 / n_batches
    dead_pct = (feature_activation_counts == 0).float().mean().item() * 100

    history["loss"].append(epoch_loss / n_batches)
    history["recon"].append(epoch_recon / n_batches)
    history["l1"].append(epoch_l1 / n_batches)
    history["l0"].append(avg_l0)
    history["dead_pct"].append(dead_pct)

    if (epoch + 1) % 5 == 0:
        print(f"Epoch {epoch+1:3d} | loss={epoch_loss/n_batches:.4f} | "
              f"recon={epoch_recon/n_batches:.4f} | L0={avg_l0:.1f} | dead={dead_pct:.1f}%")

print("Training complete.")

Step 4: Evaluate Reconstruction Quality

# Plot training curves
fig, axes = plt.subplots(1, 3, figsize=(14, 3.5))

axes[0].plot(history["recon"], color="#9c3a4a")
axes[0].set_title("Reconstruction Loss"); axes[0].set_xlabel("Epoch")

axes[1].plot(history["l0"], color="#c8832a")
axes[1].set_title("Mean L0 (active features per token)"); axes[1].set_xlabel("Epoch")

axes[2].plot(history["dead_pct"], color="#2a7c7c")
axes[2].set_title("Dead Feature %"); axes[2].set_xlabel("Epoch")

plt.tight_layout()
plt.savefig("sae_training_curves.png", dpi=150)
print("Saved: sae_training_curves.png")

# Explained variance on held-out activations
sae.eval()
with torch.no_grad():
    x_sample  = all_acts[:500].to(device)
    x_hat, h, *_ = sae(x_sample)
    var_orig  = x_sample.var(dim=0).sum().item()
    var_resid = (x_sample - x_hat).var(dim=0).sum().item()
    explained = 1 - var_resid / var_orig
    l0_final  = (h > 0).float().sum(dim=-1).mean().item()
    dead_final = (feature_activation_counts == 0).float().mean().item()

print(f"\n=== Final SAE Quality ===")
print(f"  Explained variance: {explained:.3f}  (target: >0.80)")
print(f"  Mean L0 (sparsity): {l0_final:.1f}  (target: 10–100)")
print(f"  Dead features:      {dead_final*100:.1f}%  (target: <10%)")

Step 5: Inspect Discovered Features

# For each SAE feature, find which tokens maximally activate it

# Re-collect activations with token labels this time
token_buffer  = []
act_buffer2   = []

hook2 = model.transformer.h[LAYER].mlp.register_forward_hook(
    lambda m, i, o: act_buffer2.append(o.detach().cpu()))

for text in corpus_texts[:10]:  # first 10 texts for inspection
    ids   = tok(text, return_tensors="pt", max_length=64, truncation=True)
    toks  = tok.convert_ids_to_tokens(ids["input_ids"][0])
    token_buffer.extend(toks)
    with torch.no_grad():
        model(**ids.to(device))

hook2.remove()

all_acts2  = torch.cat([a.reshape(-1, D_MODEL) for a in act_buffer2], dim=0)
sae.eval()
with torch.no_grad():
    _, h_vals, *_ = sae(all_acts2.to(device))
h_vals = h_vals.cpu()  # (N, D_SAE)

# For top 5 most-active features, show max-activating tokens
total_acts    = h_vals.sum(dim=0)         # total activation per feature
top5_features = total_acts.topk(5).indices.tolist()

print("Top 5 most active SAE features:\n")
for feat_idx in top5_features:
    feature_acts = h_vals[:, feat_idx]          # (N,)
    top10_pos    = feature_acts.topk(10).indices.tolist()
    top10_tokens = [token_buffer[p] if p < len(token_buffer) else "?"
                    for p in top10_pos]
    top10_vals   = feature_acts[top10_pos].tolist()

    print(f"Feature #{feat_idx} (total activation={total_acts[feat_idx]:.1f})")
    for tok_str, val in zip(top10_tokens, top10_vals):
        print(f"  {tok_str:20s}  act={val:.3f}")
    print()
What to look for

Some features will cluster around a single concept (e.g., all capital letters, all punctuation, or all science words). Others will be polysemantic even after SAE decomposition — this is expected with a small training corpus. A real SAE trained on billions of tokens produces much cleaner features.

Step 6: Feature Steering

Pick a feature that appears to encode a concept, then steer the model by adding its decoder direction to the MLP output at layer 9. Observe how the generated text changes.

# Choose the most active feature and steer with it
STEER_FEATURE = top5_features[0]
steer_dir     = sae.W_dec[STEER_FEATURE].detach()    # (768,) — unit vector
STEER_ALPHA   = 15.0                                 # steering strength

print(f"Steering with SAE feature #{STEER_FEATURE} (α={STEER_ALPHA})")

def steer_hook(module, inp, output):
    output[:, :, :] = output + STEER_ALPHA * steer_dir.to(device)
    return output

PROMPT = "The scientist discovered that"
input_ids = tok.encode(PROMPT, return_tensors="pt").to(device)

# Baseline generation (no steering)
with torch.no_grad():
    baseline = model.generate(
        input_ids, max_new_tokens=25, do_sample=False,
        pad_token_id=tok.eos_token_id)
print(f"Baseline:  {tok.decode(baseline[0])}")

# Steered generation
hook = model.transformer.h[LAYER].mlp.register_forward_hook(steer_hook)
with torch.no_grad():
    steered = model.generate(
        input_ids, max_new_tokens=25, do_sample=False,
        pad_token_id=tok.eos_token_id)
hook.remove()
print(f"Steered:   {tok.decode(steered[0])}")

# Try negative steering (opposite direction)
def anti_steer_hook(module, inp, output):
    output[:, :, :] = output - STEER_ALPHA * steer_dir.to(device)
    return output

hook = model.transformer.h[LAYER].mlp.register_forward_hook(anti_steer_hook)
with torch.no_grad():
    anti_steered = model.generate(
        input_ids, max_new_tokens=25, do_sample=False,
        pad_token_id=tok.eos_token_id)
hook.remove()
print(f"Anti-steer:{tok.decode(anti_steered[0])}")

Improving results

To get cleaner features: (1) collect 100× more activations using a large text dataset like WikiText-103; (2) increase expansion factor to 8× or 16×; (3) train for 50+ epochs; (4) add an auxiliary loss to prevent dead features (the "ghost grads" trick from Bricken et al. 2023).
