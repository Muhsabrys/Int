Lab 01: Tokenization Walkthrough

Track 4: Code Labs
01 of 05
~45 min
GPT-2, BERT, ViT, wav2vec2

In this lab you will: compare BPE and WordPiece tokenizers side by side, observe tokenization fragmentation on numbers and rare words, inspect special tokens, visualise ViT patch tokenization, and extract acoustic frames from wav2vec2's CNN front-end.

Before you start

Run pip install transformers torch Pillow librosa and verify you can import them. All downloads happen automatically on first run (GPT-2 ~500MB, BERT ~400MB).

Step 1: BPE Anatomy

Inspect GPT-2's BPE vocabulary. Find the longest token, the most frequent merge rule, and the token coverage of a sample sentence.

from transformers import GPT2Tokenizer
import json

tok = GPT2Tokenizer.from_pretrained("gpt2")

# 1a. Vocabulary size
print(f"Vocabulary size: {tok.vocab_size}")  # 50257

# 1b. Longest token
vocab = tok.get_vocab()
longest = max(vocab.keys(), key=lambda t: len(tok.convert_tokens_to_string([t])))
print(f"Longest token: '{longest}' ({len(longest)} chars)")

# 1c. Tokenise a test sentence
sentences = [
    "The cat sat on the mat.",
    "mechanistic interpretability",
    "uncharacteristically",
    "1234567890",
    "supercalifragilisticexpialidocious",
]
for s in sentences:
    tokens = tok.tokenize(s)
    print(f"  {s!r:45s} → {tokens}")

# 1d. Token coverage: fraction of words that are single tokens
words = "The quick brown fox jumps over the lazy dog".split()
single_token = sum(1 for w in words if len(tok.tokenize(" " + w)) == 1)
print(f"Single-token coverage: {single_token}/{len(words)} words")
Observe

Notice the leading space token (Ġ) in GPT-2. "cat" and " cat" are different tokens. BERT uses "##" prefixes instead. This affects how sentences are split.

Step 2: Compare Tokenizers

Compare GPT-2 BPE vs BERT WordPiece on the same inputs.

from transformers import GPT2Tokenizer, BertTokenizer

gpt2 = GPT2Tokenizer.from_pretrained("gpt2")
bert = BertTokenizer.from_pretrained("bert-base-uncased")

test_words = [
    "interpretability",
    "transformer",
    "2024",
    "COVID-19",
    "don't",
    "Zürich",
    "抱怨",   # Chinese characters
]

print(f"{'Word':30s} {'GPT-2':35s} {'BERT':35s}")
print("-" * 100)
for w in test_words:
    g = gpt2.tokenize(" " + w)
    b = bert.tokenize(w)
    print(f"  {w:28s} {str(g):35s} {str(b):35s}")

Step 3: Fragmentation Analysis

Measure how numbers and technical terms fragment into multiple tokens. This directly impacts model performance on arithmetic and domain-specific tasks.

import random
tok = GPT2Tokenizer.from_pretrained("gpt2")

# Number fragmentation
print("=== Number fragmentation ===")
for n in [1, 12, 123, 1234, 12345, 123456]:
    t = tok.tokenize(" " + str(n))
    print(f"  {str(n):8s} → {len(t)} token(s): {t}")

# Random number test: what fraction of 4-digit numbers are single tokens?
four_digit = [str(i) for i in range(1000, 10000)]
single = sum(1 for n in four_digit if len(tok.tokenize(" " + n)) == 1)
print(f"\n4-digit numbers that are single tokens: {single}/9000 ({100*single/9000:.1f}%)")

# Technical vocabulary fragmentation
print("\n=== Technical term fragmentation ===")
terms = ["backpropagation", "eigenvalue", "cosine",
         "hyperparameter", "stochastic", "regularisation"]
for term in terms:
    t = tok.tokenize(" " + term)
    print(f"  {term:20s} → {t}")

Step 5: ViT Patch Tokenization

import torch
from transformers import ViTModel, ViTFeatureExtractor
from PIL import Image
import numpy as np

fe = ViTFeatureExtractor.from_pretrained("google/vit-base-patch16-224")
model = ViTModel.from_pretrained("google/vit-base-patch16-224")

# Create a synthetic 224×224 RGB image
img_array = np.random.randint(0, 255, (224, 224, 3), dtype=np.uint8)
img = Image.fromarray(img_array)

# Process image
inputs = fe(images=img, return_tensors="pt")
pixel_values = inputs["pixel_values"]  # (1, 3, 224, 224)

print(f"Input image shape: {tuple(pixel_values.shape)}")
print(f"Patch size: 16×16 pixels")
print(f"Grid: {224//16}×{224//16} = {(224//16)**2} patches")
print(f"Total tokens: 196 patches + 1 [CLS] = 197")

with torch.no_grad():
    outputs = model(pixel_values=pixel_values)

hidden = outputs.last_hidden_state  # (1, 197, 768)
print(f"\nViT output shape: {tuple(hidden.shape)}")
print(f"[CLS] token norm: {hidden[0,0,:].norm():.3f}")
print(f"Mean patch norm: {hidden[0,1:,:].norm(dim=-1).mean():.3f}")

Step 6: Acoustic Frames (wav2vec2)

import torch
import numpy as np
from transformers import Wav2Vec2Model, Wav2Vec2Processor

processor = Wav2Vec2Processor.from_pretrained("facebook/wav2vec2-base-960h")
model = Wav2Vec2Model.from_pretrained("facebook/wav2vec2-base-960h")

# Synthesise a 2-second dummy audio signal at 16kHz
sr = 16000
duration = 2.0
t = np.linspace(0, duration, int(sr * duration))
audio = np.sin(2 * np.pi * 440 * t).astype(np.float32)  # 440 Hz tone

inputs = processor(audio, sampling_rate=sr, return_tensors="pt")
print(f"Raw audio samples: {inputs.input_values.shape}")
# Expected: (1, 32000)

with torch.no_grad():
    outputs = model(**inputs, output_hidden_states=True)

# CNN feature extractor output (before transformer)
cnn_feats = outputs.extract_features  # (1, T, 512)
print(f"CNN features shape: {tuple(cnn_feats.shape)}")
expected_frames = int(sr * duration / 320)  # 320 samples per frame
print(f"Expected frames: ~{expected_frames} (1 per 20ms)")
print(f"Actual frames: {cnn_feats.shape[1]}")

# Full transformer output
last = outputs.last_hidden_state  # (1, T, 768)
print(f"Transformer output: {tuple(last.shape)}")
