Lab 03: Activation Patching Walkthrough

Track 4: Code Labs
03 of 05
~75 min
GPT-2 small

This lab implements activation patching from scratch using PyTorch forward hooks. You will reproduce a simplified causal tracing experiment inspired by Meng et al. 2022 (ROME) to identify which GPT-2 layers causally store the fact "The Eiffel Tower is in Paris."

Core concept

Activation patching asks: if I replace the activation at layer L with the activation from a clean run, while keeping everything else from the corrupted run, how much does the correct answer's probability recover? Recovery = causal importance.

Setup

import torch
import numpy as np
import matplotlib.pyplot as plt
from transformers import GPT2LMHeadModel, GPT2Tokenizer

model = GPT2LMHeadModel.from_pretrained("gpt2")
tok   = GPT2Tokenizer.from_pretrained("gpt2")
model.eval()

# The fact we will trace
CLEAN_PROMPT     = "The Eiffel Tower is in"
CORRUPTED_PROMPT = "The Colosseum is in"   # different subject → Rome, not Paris
TARGET_TOKEN     = " Paris"
target_id = tok.encode(TARGET_TOKEN)[0]
print(f"Target token id: {target_id} → '{TARGET_TOKEN}'")

Step 1: Clean and Corrupted Runs

def get_logit_for_token(prompt, model, tok, token_id):
    """Return the logit for token_id at the last position."""
    ids = tok.encode(prompt, return_tensors="pt")
    with torch.no_grad():
        out = model(ids)
    logit = out.logits[0, -1, token_id].item()
    prob  = torch.softmax(out.logits[0, -1], dim=-1)[token_id].item()
    return logit, prob

clean_logit,  clean_prob  = get_logit_for_token(CLEAN_PROMPT,     model, tok, target_id)
corr_logit,   corr_prob   = get_logit_for_token(CORRUPTED_PROMPT, model, tok, target_id)

print(f"Clean run    — logit: {clean_logit:.3f}, P(' Paris'): {clean_prob:.4f}")
print(f"Corrupted run — logit: {corr_logit:.3f}, P(' Paris'): {corr_prob:.4f}")
print(f"Gap (clean - corrupted): {clean_logit - corr_logit:.3f}")

Step 2: Hook Infrastructure

# Storage for clean activations
clean_cache = {}

def make_save_hook(layer_idx, position):
    """Returns a hook that saves the residual stream at given layer/position."""
    def hook(module, input, output):
        # output is (hidden_states, present_kv, ...)
        hs = output[0] if isinstance(output, tuple) else output
        clean_cache[layer_idx] = hs.detach().clone()  # (1, T, 768)
    return hook

def make_patch_hook(layer_idx, position, cache):
    """Returns a hook that patches position `position` from cache."""
    def hook(module, input, output):
        hs = output[0] if isinstance(output, tuple) else output
        hs = hs.clone()
        hs[0, position, :] = cache[layer_idx][0, position, :]
        if isinstance(output, tuple):
            return (hs,) + output[1:]
        return hs
    return hook

# Step 2a: Save clean activations for all 12 layers
clean_ids = tok.encode(CLEAN_PROMPT, return_tensors="pt")
last_pos   = clean_ids.shape[1] - 1

hooks = []
for l in range(12):
    h = model.transformer.h[l].register_forward_hook(make_save_hook(l, last_pos))
    hooks.append(h)

with torch.no_grad():
    model(clean_ids)

for h in hooks:
    h.remove()

print(f"Saved clean activations for {len(clean_cache)} layers.")
print(f"Each activation shape: {clean_cache[0].shape}")

Step 3: Layer-by-Layer Patching

corr_ids = tok.encode(CORRUPTED_PROMPT, return_tensors="pt")
# Note: corrupted prompt may be shorter — we patch last position
corr_last = corr_ids.shape[1] - 1

patch_results = []

for patch_layer in range(12):
    # Install patch hook only on patch_layer
    hook = model.transformer.h[patch_layer].register_forward_hook(
        make_patch_hook(patch_layer, corr_last, clean_cache)
    )

    with torch.no_grad():
        out = model(corr_ids)

    hook.remove()

    patched_logit = out.logits[0, -1, target_id].item()

    # Normalised recovery: 0 = no recovery, 1 = full recovery
    recovery = (patched_logit - corr_logit) / (clean_logit - corr_logit + 1e-9)
    patch_results.append(recovery)
    print(f"  Patch layer {patch_layer:2d}: logit={patched_logit:.3f}, recovery={recovery:.3f}")

print("\nTop layers by causal effect:")
for l, r in sorted(enumerate(patch_results), key=lambda x: -x[1])[:3]:
    print(f"  Layer {l}: recovery={r:.3f}")

fig, ax = plt.subplots(figsize=(10, 3.5))

Step 5: Attention Head Patching

# Patch individual attention heads within the top causal layer
top_layer = int(np.argmax(patch_results))
print(f"Top causal layer: {top_layer}")

# Save attention output for each head in top_layer
head_cache = {}

def save_attn_hook(module, input, output):
    # output[0] is attention output (1, T, 768)
    head_cache["attn_out"] = output[0].detach().clone()

hook = model.transformer.h[top_layer].attn.register_forward_hook(save_attn_hook)
with torch.no_grad():
    model(clean_ids)
hook.remove()

# Patch each of the 12 heads individually
d_head = 768 // 12  # 64 dims per head
head_recoveries = []

for head_idx in range(12):
    start = head_idx * d_head
    end   = start + d_head

    def patch_head_hook(module, input, output, _h=head_idx):
        out = output[0].clone()
        out[0, corr_last, _h*d_head:(_h+1)*d_head] = \
            head_cache["attn_out"][0, last_pos, _h*d_head:(_h+1)*d_head]
        return (out,) + output[1:]

    hook = model.transformer.h[top_layer].attn.register_forward_hook(patch_head_hook)
    with torch.no_grad():
        out = model(corr_ids)
    hook.remove()

    logit = out.logits[0, -1, target_id].item()
    rec   = (logit - corr_logit) / (clean_logit - corr_logit + 1e-9)
    head_recoveries.append(rec)
    print(f"  Layer {top_layer}, Head {head_idx:2d}: recovery={rec:.3f}")

Step 6: MLP vs Attention Contribution

# For the top causal layer, compare patching the full layer,
# just the MLP, and just the attention

mlp_cache = {}
attn_cache_full = {}

def save_mlp_hook(module, input, output):
    mlp_cache["mlp"] = output.detach().clone()

def save_attn_full_hook(module, input, output):
    attn_cache_full["attn"] = output[0].detach().clone()

h1 = model.transformer.h[top_layer].mlp.register_forward_hook(save_mlp_hook)
h2 = model.transformer.h[top_layer].attn.register_forward_hook(save_attn_full_hook)
with torch.no_grad():
    model(clean_ids)
h1.remove(); h2.remove()

# Patch only MLP output
def patch_mlp_hook(module, input, output):
    out = output.clone()
    out[0, corr_last, :] = mlp_cache["mlp"][0, last_pos, :]
    return out

hook = model.transformer.h[top_layer].mlp.register_forward_hook(patch_mlp_hook)
with torch.no_grad():
    out_mlp = model(corr_ids)
hook.remove()
mlp_logit = out_mlp.logits[0, -1, target_id].item()
mlp_rec   = (mlp_logit - corr_logit) / (clean_logit - corr_logit + 1e-9)

print(f"\nLayer {top_layer} breakdown:")
print(f"  Full layer patch: recovery={patch_results[top_layer]:.3f}")
print(f"  MLP only:        recovery={mlp_rec:.3f}")
print(f"  Attn only:       recovery≈{patch_results[top_layer] - mlp_rec:.3f} (approx)")

Interpretation note

GPT-2 small is a small model — results will vary from large models (GPT-2 XL, GPT-4). The key takeaway is the method, not the exact layer numbers. In larger models, factual recall tends to localise to later MLP layers.
