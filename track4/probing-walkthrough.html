Lab 02: Probing Walkthrough

Track 4: Code Labs
02 of 05
~60 min
BERT-base-uncased

In this lab you will extract hidden states from all 12 BERT layers, build a simple dataset of token-level linguistic properties (part-of-speech, capitalisation, position), train a linear probe on each layer's representations, and visualise which layers encode which properties. You will also implement a control probe to check that high accuracy isn't trivially explained by input statistics.

Setup

# Required packages
# pip install transformers torch scikit-learn matplotlib numpy

import torch
import numpy as np
from transformers import BertModel, BertTokenizerFast
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score
from sklearn.preprocessing import StandardScaler
import matplotlib.pyplot as plt

tok   = BertTokenizerFast.from_pretrained("bert-base-uncased")
model = BertModel.from_pretrained("bert-base-uncased")
model.eval()
print("BERT loaded. Layers: 12, Hidden dim: 768")

Step 1: Extract Hidden States

def get_all_layer_hidden_states(sentences, tokenizer, model, max_len=64):
    """
    Returns: list of (token_str, layer_hiddens) where
             layer_hiddens is shape (13, 768) — one vector per layer
    """
    all_tokens = []
    all_hiddens = []  # list of (13, 768) arrays

    for sent in sentences:
        enc = tokenizer(sent, return_tensors="pt",
                        max_length=max_len, truncation=True,
                        return_offsets_mapping=True)
        with torch.no_grad():
            out = model(
                input_ids=enc["input_ids"],
                attention_mask=enc["attention_mask"],
                output_hidden_states=True
            )

        # hidden_states: tuple of 13 tensors, each (1, T, 768)
        hs = torch.stack(out.hidden_states, dim=0)  # (13, 1, T, 768)
        hs = hs[:, 0, :, :]                          # (13, T, 768)

        tokens = tokenizer.convert_ids_to_tokens(enc["input_ids"][0])
        for pos, token in enumerate(tokens):
            if token in ["[CLS]", "[SEP]", "[PAD]"]:
                continue
            all_tokens.append(token)
            all_hiddens.append(hs[:, pos, :].numpy())  # (13, 768)

    return all_tokens, np.stack(all_hiddens)  # (N, 13, 768)

# Sample sentences covering diverse POS tags
sentences = [
    "The quick brown fox jumps over the lazy dog.",
    "Scientists discovered a new species of deep-sea fish.",
    "Paris is the capital and most populous city of France.",
    "She carefully placed the fragile vase on the wooden shelf.",
    "Running fast is difficult when carrying heavy equipment.",
    "The transformer architecture was introduced in 2017.",
    "He said that he would never return to that place.",
    "Mathematical proofs require rigorous logical reasoning.",
]
tokens, hiddens = get_all_layer_hidden_states(sentences, tok, model)
print(f"Extracted {len(tokens)} token positions, {hiddens.shape[1]} layers, dim={hiddens.shape[2]}")

Step 2: Build Probe Dataset

# Create labels for three probing tasks

# Task A: Is the token a capitalised word-piece?
labels_cap = np.array([1 if t[0].isupper() else 0 for t in tokens])

# Task B: Is the token a subword (starts with ##)?
labels_sub = np.array([1 if t.startswith("##") else 0 for t in tokens])

# Task C: Rough POS — is the token a common function word?
function_words = {"the", "a", "an", "is", "are", "was", "of",
                  "in", "on", "and", "or", "that", "he", "she",
                  "it", "they", "to", "for", "with", "at"}
labels_func = np.array([1 if t.lower() in function_words else 0
                        for t in tokens])

print(f"Capitalised tokens: {labels_cap.sum()} / {len(labels_cap)}")
print(f"Subword tokens: {labels_sub.sum()} / {len(labels_sub)}")
print(f"Function word tokens: {labels_func.sum()} / {len(labels_func)}")

Step 3: Train Probes on Each Layer

def probe_all_layers(hiddens, labels, task_name):
    """Train a logistic regression probe on each of 13 layers.
    Returns: list of 13 cross-validation accuracy scores."""
    n_layers = hiddens.shape[1]
    scores = []
    for l in range(n_layers):
        X = hiddens[:, l, :]          # (N, 768)
        X = StandardScaler().fit_transform(X)
        clf = LogisticRegression(max_iter=200, C=0.1)
        cv_scores = cross_val_score(clf, X, labels, cv=3, scoring="accuracy")
        scores.append(cv_scores.mean())
        print(f"  [{task_name}] Layer {l:2d}: {cv_scores.mean():.3f}")
    return scores

print("Probing for capitalisation...")
cap_scores  = probe_all_layers(hiddens, labels_cap,  "capitalisation")

print("Probing for subword...")
sub_scores  = probe_all_layers(hiddens, labels_sub,  "subword")

print("Probing for function words...")
func_scores = probe_all_layers(hiddens, labels_func, "function-word")

Step 5: Control Probes

A control probe uses a randomly shuffled label vector. If the control probe achieves near-chance accuracy but the real probe doesn't, that's evidence the real probe is using genuine information. If both are high, there's a data leakage issue.

import random

# Shuffle labels to create a random control
np.random.seed(42)
labels_cap_shuffled = labels_cap.copy()
np.random.shuffle(labels_cap_shuffled)

print("Control probe (shuffled capitalisation labels):")
ctrl_scores = probe_all_layers(hiddens, labels_cap_shuffled, "control")

# Compare: real probe should beat control by a clear margin
print("\n=== Real vs Control ===")
for l in range(13):
    delta = cap_scores[l] - ctrl_scores[l]
    print(f"  Layer {l:2d}: real={cap_scores[l]:.3f}, control={ctrl_scores[l]:.3f}, Δ={delta:+.3f}")

Step 6: MDL (Minimum Description Length) Probe

MDL probing measures not just accuracy but how much compression the hidden state provides. A probe that learns quickly (uses few samples) has lower MDL — it finds more genuine information. This addresses the criticism that complex probes can memorise even random labels.

from sklearn.linear_model import LogisticRegression
from sklearn.metrics import log_loss
import numpy as np

def online_code_length(X, y, n_splits=10):
    """Approximate MDL via online coding: train on first k samples,
    evaluate loss on next k samples, sum over all splits."""
    N = len(y)
    chunk = N // n_splits
    total_bits = 0.0
    for i in range(1, n_splits):
        train_X, train_y = X[:i*chunk], y[:i*chunk]
        test_X,  test_y  = X[i*chunk:(i+1)*chunk], y[i*chunk:(i+1)*chunk]
        clf = LogisticRegression(max_iter=200, C=0.1).fit(train_X, train_y)
        probs = clf.predict_proba(test_X)
        bits = log_loss(test_y, probs, normalize=False) / np.log(2)
        total_bits += bits
    return total_bits

# Compute code length for capitalisation at each layer
print("MDL code length per layer (lower = more informative):")
for l in range(13):
    X = StandardScaler().fit_transform(hiddens[:, l, :])
    cl = online_code_length(X, labels_cap)
    print(f"  Layer {l:2d}: {cl:.1f} bits")
