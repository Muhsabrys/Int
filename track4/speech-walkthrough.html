Lab 04: Speech Model Walkthrough

Track 4: Code Labs
04 of 05
~80 min
wav2vec2-base, Whisper-tiny

This lab walks through extracting and analysing representations from two major open-source speech models. You will visualise the spectrogram, extract layer-wise features from wav2vec2, run a phoneme-level linear probe, transcribe audio with Whisper, and visualise the cross-attention alignment pattern.

Audio files

This lab uses the datasets library to load a small speech sample from LibriSpeech. No local audio files are needed — everything downloads automatically. If you have a WAV file, you can substitute it wherever audio["array"] appears.

Setup

# pip install transformers torch datasets librosa soundfile matplotlib numpy

import torch
import numpy as np
import matplotlib.pyplot as plt
from transformers import (
    Wav2Vec2Model, Wav2Vec2Processor,
    WhisperForConditionalGeneration, WhisperProcessor
)
from datasets import load_dataset

# Load one sample from LibriSpeech test-clean
ds = load_dataset("hf-internal-testing/librispeech_asr_demo",
                  "clean", split="validation", trust_remote_code=True)
sample = ds[0]
audio  = sample["audio"]
sr     = audio["sampling_rate"]  # 16000
waveform = np.array(audio["array"], dtype=np.float32)

print(f"Audio duration: {len(waveform)/sr:.2f}s at {sr}Hz")
print(f"Ground truth: '{sample['text']}'")

Step 1: Visualise the Waveform and Spectrogram

import librosa
import librosa.display

fig, axes = plt.subplots(2, 1, figsize=(12, 5))

# Waveform
axes[0].plot(np.linspace(0, len(waveform)/sr, len(waveform)),
              waveform, color="#c8832a", linewidth=0.5)
axes[0].set_xlabel("Time (s)")
axes[0].set_ylabel("Amplitude")
axes[0].set_title("Waveform")

# Log-Mel spectrogram
mel = librosa.feature.melspectrogram(y=waveform, sr=sr, n_mels=80)
mel_db = librosa.power_to_db(mel, ref=np.max)
librosa.display.specshow(mel_db, sr=sr, x_axis="time", y_axis="mel", ax=axes[1])
axes[1].set_title("80-channel Log-Mel Spectrogram")

plt.tight_layout()
plt.savefig("speech_spectrogram.png", dpi=150)
print("Saved: speech_spectrogram.png")

Step 2: Extract wav2vec2 Layer Features

processor = Wav2Vec2Processor.from_pretrained("facebook/wav2vec2-base-960h")
w2v_model = Wav2Vec2Model.from_pretrained("facebook/wav2vec2-base-960h")
w2v_model.eval()

inputs = processor(waveform, sampling_rate=sr, return_tensors="pt")

with torch.no_grad():
    outputs = w2v_model(**inputs, output_hidden_states=True)

# outputs.hidden_states: tuple of 13 tensors (1, T, 768)
# outputs.extract_features: CNN output before transformer (1, T, 512)

print(f"Number of frames: {outputs.last_hidden_state.shape[1]}")
print(f"Frame rate: ~{outputs.last_hidden_state.shape[1] / (len(waveform)/sr):.0f} fps")
print(f"Hidden dim: {outputs.last_hidden_state.shape[2]}")
print(f"Layers available: {len(outputs.hidden_states)} (incl. embedding)")

# Visualise layer-wise mean activation norm
norms = [outputs.hidden_states[l][0].norm(dim=-1).mean().item()
         for l in range(len(outputs.hidden_states))]
plt.figure(figsize=(8, 3))
plt.plot(norms, "o-", color="#2a7c7c")
plt.xlabel("Layer"); plt.ylabel("Mean L2 norm")
plt.title("wav2vec2 Residual Stream Norm by Layer")
plt.savefig("w2v_norms.png", dpi=150)
print("Saved: w2v_norms.png")

Step 3: Phoneme-Level Linear Probe

from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score
from sklearn.preprocessing import StandardScaler

# Synthetic phoneme labels: voiced (1) vs voiceless (0)
# Voiced: a,e,i,o,u,b,d,g,v,z,m,n,l,r
# We approximate by labeling frames by energy quintile
# (high energy ≈ vowels/voiced, low energy ≈ silences/voiceless)
energy = waveform ** 2
# Resample energy to match number of frames
n_frames = outputs.last_hidden_state.shape[1]
energy_ds = np.array([
    energy[int(i * len(energy) / n_frames) : int((i+1) * len(energy) / n_frames)].mean()
    for i in range(n_frames)
])
labels = (energy_ds > np.median(energy_ds)).astype(int)

# Probe each layer
print("Probing for high-energy frames (proxy for voiced):")
layer_scores = []
for l in range(len(outputs.hidden_states)):
    X = outputs.hidden_states[l][0].numpy()  # (T, 768)
    X = StandardScaler().fit_transform(X)
    clf = LogisticRegression(max_iter=300, C=0.5)
    scores = cross_val_score(clf, X, labels, cv=3)
    layer_scores.append(scores.mean())
    print(f"  Layer {l:2d}: {scores.mean():.3f}")

plt.figure(figsize=(8, 3))
plt.plot(layer_scores, "s-", color="#9c3a4a")
plt.xlabel("Layer"); plt.ylabel("Probe Accuracy")
plt.title("wav2vec2 Layer-wise Probe (voiced vs voiceless proxy)")
plt.savefig("w2v_probing.png", dpi=150)
print("Saved: w2v_probing.png")

Step 4: Transcribe with Whisper

whisper_proc  = WhisperProcessor.from_pretrained("openai/whisper-tiny")
whisper_model = WhisperForConditionalGeneration.from_pretrained("openai/whisper-tiny")
whisper_model.eval()

features = whisper_proc(waveform, sampling_rate=sr, return_tensors="pt").input_features

with torch.no_grad():
    predicted_ids = whisper_model.generate(features)

transcript = whisper_proc.batch_decode(predicted_ids, skip_special_tokens=True)[0]
print(f"Whisper transcript: '{transcript}'")
print(f"Ground truth:       '{sample['text']}'")

Step 5: Visualise Cross-Attention Alignment

# Re-run with output_attentions=True to capture cross-attention
with torch.no_grad():
    out_w_attn = whisper_model(
        input_features=features,
        decoder_input_ids=predicted_ids[:, :-1],
        output_attentions=True
    )

# Cross-attentions: list of (n_layers,) each (1, n_heads, T_text, T_audio)
cross_attns = out_w_attn.cross_attentions  # available in encoder-decoder models

if cross_attns is not None:
    # Average across heads, pick middle decoder layer
    mid_layer = len(cross_attns) // 2
    avg_attn = cross_attns[mid_layer][0].mean(dim=0).numpy()  # (T_text, T_audio)

    fig, ax = plt.subplots(figsize=(12, 4))
    im = ax.imshow(avg_attn, aspect="auto", origin="lower", cmap="inferno")
    ax.set_xlabel("Audio frame")
    ax.set_ylabel("Output text token")
    ax.set_title(f"Whisper Cross-Attention (decoder layer {mid_layer}, avg heads)")
    plt.colorbar(im)
    plt.tight_layout()
    plt.savefig("whisper_cross_attention.png", dpi=150)
    print("Saved: whisper_cross_attention.png")
else:
    print("Cross-attention not available for generate() output. Use forward() directly.")

Step 6: Speaker Direction via PCA

from sklearn.decomposition import PCA

# Use first 3 wav2vec2 samples from the dataset to find speaker direction
all_layer1 = []
for i in range(min(3, len(ds))):
    wav_i = np.array(ds[i]["audio"]["array"], dtype=np.float32)
    inp_i = processor(wav_i, sampling_rate=sr, return_tensors="pt")
    with torch.no_grad():
        out_i = w2v_model(**inp_i, output_hidden_states=True)
    # Take layer-1 representations, mean over time
    rep = out_i.hidden_states[1][0].mean(dim=0).numpy()  # (768,)
    all_layer1.append(rep)

all_layer1 = np.stack(all_layer1)  # (3, 768)

# For a real study you'd need many more samples per speaker
# Here we just show the PCA directions
pca = PCA(n_components=2).fit(all_layer1)
print(f"PCA explained variance ratio: {pca.explained_variance_ratio_}")
print(f"PC1 norm: {np.linalg.norm(pca.components_[0]):.3f}")
print("(In a full study with many speakers, PC1 would correlate with speaker identity)")

Stretch Goals
